create table tmembers (
    idx int primary key auto_increment not null,
    tu_check char(1),
    tu_name varchar(50) not null,
    tu_gender char(1) not null,
    tu_birth varchar(8) not null,
    tu_mail varchar(60) not null,
    tu_pwd varchar(16) not null,
    tu_reg TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    tu_edit date
);
